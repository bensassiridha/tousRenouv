            function GetScreenBounds() {
                var bounds = new Object();
                bounds.width =
                    parseInt(window.innerWidth || document.documentElement.clientWidth || document.body.clientWidth || 0);
                bounds.height =
                    parseInt(window.innerHeight || document.documentElement.clientHeight || document.body.clientHeight || 0);
                return bounds;

            function GetScrollOffset() {
                var bounds = new Object();

                bounds.y = parseInt(window.pageYOffset || document.body.scrollTop || document.documentElement.scrollTop || 0);

                return bounds;
